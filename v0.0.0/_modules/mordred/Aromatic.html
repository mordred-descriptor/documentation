from ._base import Descriptor

__all__ = (
    'AromaticAtomsCount', 'AromaticBondsCount',
)


class AromaticBase(Descriptor):
    __slots__ = ()

    @classmethod
    def preset(cls):
        yield cls()

    def __str__(self):
        return self._name

    def parameters(self):
        return ()

    rtype = int



[docs]class AromaticAtomsCount(AromaticBase):
    r"""aromatic atoms count descriptor."""
    __slots__ = ()
    _name = 'nAromAtom'

    def calculate(self):
        return sum(
            1
            for a in self.mol.GetAtoms()
            if a.GetIsAromatic()
        )
