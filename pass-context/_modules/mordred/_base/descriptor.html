    def __repr__(self):
        cls, args = self.as_key()
        return '{}({})'.format(cls.__name__, ', '.join(self._pretty(a) for a in args))

    def __hash__(self):
        return hash(self.as_key())

    def __compare_by_reduce(meth):
        def compare(self, other):
            l = self.as_key()
            r = other.as_key()
            return getattr(l, meth)(r)

        return compare
